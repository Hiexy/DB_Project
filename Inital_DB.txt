CREATE TABLE Floor (  
    ID int NOT NULL PRIMARY KEY, 
    Foot_Traffic FLOAT(15), 
    Name VARCHAR(255) NOT NULL 
     
);

CREATE TABLE Business (  
    ID int NOT NULL PRIMARY KEY,  
    Phone_Number VARCHAR(10),  
    Location VARCHAR(255),  
    Name VARCHAR(255) NOT NULL,  
    Clflag NUMBER(1),  
    Suflag NUMBER(1),  
    Bflag NUMBER(1),  
    Eflag NUMBER(1),  
    Seflag NUMBER(1),  
    Rflag NUMBER(1),  
    Fflag NUMBER(1),  
    Coflag NUMBER(1)   
);

CREATE TABLE Store (  
    ID int NOT NULL PRIMARY KEY, 
    Location VARCHAR(2) NOT NULL, 
    Contract_Start DATE,  
    Contract_End DATE, 
    Name VARCHAR(5) NOT NULL, 
    Space int NOT NULL, 
    Rented NUMBER(1), 
    Visibility FLOAT(15) NOT NULL, 
    Sales FLOAT(15), 
    Rental_Fee FLOAT(15), 
    Floor_ID int, 
    Business_ID int 
);
